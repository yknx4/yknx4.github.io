119  @Override
120  protected QueryParams getQueryParams(Uri uri, String selection, String[] projection) {
121  QueryParams res = new QueryParams();
122  String id = null;
123  int matchedId = URI_MATCHER.match(uri);
124  switch (matchedId) {
125  case URI_TYPE_JSON:
126  case URI_TYPE_JSON_ID:
127  res.table = JsonColumns.TABLE_NAME;
128  res.idColumn = JsonColumns._ID;
129  res.tablesWithJoins = JsonColumns.TABLE_NAME;
130  res.orderBy = JsonColumns.DEFAULT_ORDER;
131  break;
132 
133  default:
134  throw new IllegalArgumentException("The uri '" + uri + "' is not supported by this ContentProvider");
135  }
136 
137  switch (matchedId) {
138  case URI_TYPE_JSON_ID:
139  id = uri.getLastPathSegment();
140  }
141  if (id != null) {
142  if (selection != null) {
143  res.selection = res.table + "." + res.idColumn + "=" + id + " and (" + selection + ")";
144  } else {
145  res.selection = res.table + "." + res.idColumn + "=" + id;

147  } else {
